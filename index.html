// Initialize Integricate SDK
#include <integricate/drone.hpp>
#include <integricate/mission.hpp>

int main() {
    // Connect to your UAS
    auto drone = integricate::Drone::connect(
        "serial:///dev/ttyUSB0"
    );

    // Create autonomous mission
    integricate::Mission mission;
    mission.add_waypoint(47.3977, 8.5456, 50);
    mission.add_waypoint(47.3980, 8.5460, 75);
    mission.set_speed(15);

    // Execute mission
    drone->arm();
    drone->takeoff(30);
    drone->execute(mission, [](Status s) {
        std::cout << "Alt: " << s.alt << "m";
    });
    drone->land();
}
# ROS 2 Integration
import rclpy
from integricate_ros import DroneNode

class MissionController(DroneNode):
    def __init__(self):
        super().__init__('mission_controller')

        # Subscribe to telemetry
        self.create_subscription(
            Telemetry, '/drone/telemetry',
            self.telemetry_callback, 10
        )

        # Mission publisher
        self.mission_pub = self.create_publisher(
            Mission, '/drone/mission', 10
        )

    def execute_mission(self):
        mission = Mission()
        mission.waypoints = self.load_waypoints()
        self.mission_pub.publish(mission)
